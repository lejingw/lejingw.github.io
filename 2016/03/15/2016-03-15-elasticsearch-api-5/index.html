ElasticSearch API 5-结构化搜索

Posted at 2016-03-15 technology  elasticsearch 

结构化搜索

构建测试数据

1 创建索引
curl -XDELETE -i 'http://localhost:9200/my_store'
curl -XPUT -i "http://127.0.0.1:9200/my_store" -d '
{
   "mappings":{
     "products":{
        "properties":{
          "productID":{
            "type":"string",
            "index":"not_analyzed"
           }
        }
      }
    }
}'

2 批量添加数据
curl -XPOST -i 'http://localhost:9200/my_store/products/_bulk' -d '
{ "index": { "_id": 1 }}
{ "price" : 10, "productID" : "XHDK-A-1293-#fJ3" }
{ "index": { "_id": 2 }}
{ "price" : 20, "productID" : "KDKE-B-9947-#kL5" }
{ "index": { "_id": 3 }}
{ "price" : 30, "productID" : "JODL-X-1937-#pV7" }
{ "index": { "_id": 4 }}
{ "price" : 30, "productID" : "QQPX-R-3956-#aD8" }
'

3 准确值查询
curl -XGET -i 'http://localhost:9200/my_store/products/_search' -d '{
    "query":{
        "filtered":{
            "filter":{
                "term":{
                    "productID":"KDKE-B-9947-#kL5"
                }
            }
        }
    }
}'

组合过滤

curl -XGET -i 'http://localhost:9200/my_store/products/_search' -d '{
"query" : {
        "filtered" : {
            "filter" : {
                "bool":{
                    "should":[
                        {"term":{"price":20}},
                        {"term":{"procudtID":"XHDK-A-1293-#fJ3"}}
                    ],
                    "must_not":{
                        "term":{"price":30}
                    }
                }
            }
        }
    }
}'

SELECT product FROM products
WHERE (price = 20 OR productID = “XHDK-A-1293-#fJ3”)
AND (price != 30)

查询多个准确值

{
    "terms":{
        "price":[20,30]
    }
}

term 和 terms 是包含操作，而不是相等操作

完全匹配

假如你真的需要完全匹配这种行为，最好是通过添加另一个字段来实现。在这个字段中，你索引原字段包含值的个数。

{ "tags" : ["search"], "tag_count" : 1 }
{ "tags" : ["search", "open_source"], "tag_count" : 2 }

GET /my_index/my_type/_search
{
    "query": {
        "filtered" : {
            "filter" : {
                 "bool" : {
                    "must" : [
                        { "term" : { "tags" : "search" } },
                        { "term" : { "tag_count" : 1 } }
                    ]
                }
            }
        }
    }
}

范围

GET /my_store/products/_search
{
    "query" : {
        "filtered" : {
            "filter" : {
                "range" : {
                    "price" : {
                        "gte" : 20,
                        "lt"  : 40
                    }
                }
            }
        }
    }
}

日期范围

"range" : {
    "timestamp" : {
        "gt" : "2014-01-01 00:00:00",
        "lt" : "2014-01-07 00:00:00"
    }
}

"range":{
    "timestamp":{
        "gt":"now-1h"#最近一小时
    }
}

"range" : {
    "timestamp" : {
        "gt" : "2014-01-01 00:00:00",
        "lt" : "2014-01-01 00:00:00||+1M"#早于2014年1月1号加一个月
    }
}

只要在日期后加上双竖线||，就能使用日期数学表达式了

处理null值

GET /my_index/posts/_search
{
    "query" : {
        "filtered" : {
            "filter" : {
                "exists" : { "field" : "tags" }
            }
        }
    }
}

GET /my_index/posts/_search
{
    "query" : {
        "filtered" : {
            "filter": {
                "missing" : { "field" : "tags" }
            }
        }
    }
}

过滤顺序

bool 条件中过滤器的顺序对性能有很大的影响

  • 更详细的过滤条件应该被放置在其他过滤器之前，以便在更早的排除更多的文档
  • 缓存的过滤器非常快，所以它们需要被放在不能缓存的过滤器之前
